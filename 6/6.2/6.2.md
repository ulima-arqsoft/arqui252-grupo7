# 6.2. Diagrama de Contenedores

```mermaid
flowchart TB
%% =========================
%% Diagrama de Contenedores - LOOP (Ecommerce de Ropa)
%% =========================

classDef sys fill:#eaf4ff,stroke:#1e88e5,stroke-width:1.2px,color:#0d47a1;
classDef data fill:#fff7e6,stroke:#fb8c00,color:#e65100;
classDef cache fill:#f3e5f5,stroke:#8e24aa,color:#4a148c;

%% Contenedores Frontend (Clientes y Administración)
subgraph Frontend["Frontend"]
  A["App Android (Kotlin + Jetpack Compose)"]:::mobile
  I["App iOS (Swift + SwiftUI)"]:::mobile
  W["Web Admin (React.js)"]:::web
end

%% Contenedor Backend (.NET API C#)
subgraph Backend["Backend (.NET API C#)"]
  B["API REST ASP.NET Core"]:::sys
  subgraph Modules["Módulos Lógicos"]
    M1["Visualización de Productos"]
    M2["Gestión de Compras"]
    M3["Autenticación y Usuarios"]
    M4["Gestión de Negocio"]
  end
end

%% Base de Datos y Caching
DB["PostgreSQL (Base de Datos Relacional)"]:::data
R["Redis (Caché en Memoria)"]:::cache

%% Servicios Externos y API Integraciones
PAY["Pasarela de Pagos (Stripe / PayPal / MercadoPago)"]:::sys
OAUTH["Proveedores OAuth (Google / Apple / Facebook)"]:::sys
PUSH["Notificaciones Push (FCM / APNs)"]:::sys
OBS["Observabilidad (OpenTelemetry + Dashboards)"]:::sys

%% Relaciones
Frontend -->|Interacción| A
Frontend -->|Interacción| I
Frontend -->|Interacción| W
A -->|Comunicación API| B
I -->|Comunicación API| B
W -->|Comunicación API| B

B -->|Acceso a datos| DB
B -->|Lecturas rápidas| R
B -->|Autenticación y autorización| OAUTH
B -->|Notificaciones Push| PUSH
B -->|Métricas y logs| OBS
B -->|Procesamiento de pagos| PAY

%% Contenedores de Datos
DB -->|Almacena| M1
DB -->|Almacena| M2
DB -->|Almacena| M3
DB -->|Almacena| M4
R -->|Cachea| M1
R -->|Cachea| M2
R -->|Cachea| M3
R -->|Cachea| M4
```
